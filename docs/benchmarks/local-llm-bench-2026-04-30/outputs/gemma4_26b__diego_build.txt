// FILE: force-app/main/default/classes/VehicleTriggerHandler.cls
public with sharing class VehicleTriggerHandler {

    /**
     * Validates state machine transitions on insert.
     * Allowed: Available -> Rented, Available -> In Maintenance, Available -> Retired
     * Note: For simplicity in this implementation, we assume 'Available' is the default starting state.
     */
    public void beforeInsert(List<Vehicle__c> newRecords) {
        validateStatusTransition(newRecords, null);
    }

    /**
     * Validates state machine transitions on update.
     */
    public void beforeUpdate(List<Vehicle__c> newRecords, Map<Id, Vehicle__c> oldMap) {
        validateStatusTransition(newRecords, oldMap);
    }

    /**
     * Handles post-save logic: Creating Maintenance records and verifying Rental Contracts.
     */
    public void afterUpdate(List<Vehicle__ct> newRecords, Map<Id, Vehicle__c> oldMap) {
        handleMaintenanceCreation(newRecords, oldMap);
        handleRentalContractVerification(newRecords, oldMap);
    }

    private void validateStatusTransition(List<Vehicle__c> newRecords, Map<Id, Vehicle__c> oldMap) {
        for (Vehicle__c vehicle : newRecords) {
            String newStatus = vehicle.Condition_Status__c;
            String oldStatus = (oldMap != null && oldMap.containsKey(vehicle.Id)) ? oldMap.get(vehicle.Id).Condition_Status__c : 'Available';

            if (newStatus == oldStatus) continue;

            // State Machine Logic
            // Allowed: Available -> Rented, Available -> In Maintenance, Available -> Retired
            // Also allow transitions from other states if they follow the logic (e.g., In Maintenance -> Available)
            Boolean isValid = false;

            if (oldStatus == 'Available') {
                if (newStatus == 'Rented' || newStatus == 'In Maintenance' || newStatus == 'Out of Service') {
                    isValid = true;
                }
            } else if (oldStatus == 'In Maintenance' || oldStatus == 'Rented' || oldStatus == 'Out of Service') {
                if (newStatus == 'Available') {
                    isValid = true;
                }
            } else if (newStatus == 'Retired') {
                isValid = true; // Can retire from any state
            }

            if (!isValid) {
                throw new AuraHandledException('Invalid status transition from ' + oldStatus + ' to ' + newStatus);
            }
        }
    }

    private void handleMaintenanceCreation(List<Vehicle__c> newRecords, Map<Id, Vehicle__c> oldMap) {
        List<Maintenance__c> maintenanceToCreate = new List<Maintenance__c>();
        
        for (Vehicle__c vehicle : newRecords) {
            String newStatus = vehicle.Condition_Status__c;
            String oldStatus = (oldMap != null && oldMap.containsKey(vehicle.Id)) ? oldMap.get(vehicle.Id).Condition_Status__c : null;

            // Check if status changed to 'In Maintenance'
            // Note: Using 'In Maintenance' as per requirement, though field value is 'In Maintenance'
            if (newStatus == 'In Maintenance' && (oldStatus != 'In Maintenance')) {
                maintenanceToCreate.add(new Maintenance__c(
                    Vehicle__c = vehicle.Id,
                    Maintenance_Date__c = Date.today(),
                    Description__c = 'Automated maintenance record created due to status change.'
                ));
            }
        }

        if (!maintenanceToCreate.isEmpty()) {
            try {
                insert as user maintenanceToCreate;
            } catch (DmlException e) {
                throw new AuraHandledException('Failed to create maintenance record: ' + e.getMessage());
            }
        }
    }

    private void handleRentalContractVerification(List<Vehicle__c> newRecords, Map<Id, Vehicle__c> oldMap) {
        Set<Id> vehicleIdsToVerify = new Set<Id>();

        for (Vehicle__c vehicle : newRecords) {
            String newStatus = vehicle.Condition_Status__c;
            String oldStatus = (oldMap != null && oldMap.containsKey(vehicle.Id)) ? oldMap.get(vehicle.Id).Condition_Status__c : null;

            if (newStatus == 'Rented' && oldStatus != 'Rented') {
                vehicleIdsToVerify.add(vehicle.Id);
            }
        }

        if (!vehicleIdsToVerify.isEmpty()) {
            // Query for active RentalContract__c records
            // Assuming RentalContract__c is a custom object related to Vehicle__c
            // Since RentalContract__c wasn't in the provided JSON, I will implement the logic 
            // assuming a standard relationship exists as per the prompt's requirement.
            
            // We use a Map to track which vehicles have contracts
            Map<Id, Boolean> vehicleHasContract = new Map<Id, Boolean>();
            for(Id vId : vehicleIdsToVerify) vehicleHasContract.put(vId, false);

            try {
                // Note: Using a hypothetical RentalContract__c object as per requirement 3
                // We check for 'Active' status.
                for (RentalContract__c contract : [
                    SELECT Vehicle__c 
                    FROM RentalContract__c 
                    WHERE Vehicle__c IN :vehicleIdsToVerify 
                    AND Status__c = 'Active' 
                    WITH SECURITY_ENFORCED
                ]) {
                    vehicleHasContract.put(contract.Vehicle__c, true);
                }
            } catch (Exception e) {
                // If object doesn't exist in environment, we catch to prevent deployment failure in partial environments
                // but in a real scenario, this object must exist.
            }

            for (Vehicle__c vehicle : newRecords) {
                if (vehicleIdsToVerify.contains(vehicle.Id) && vehicleHasContract.get(vehicle.Id) == false) {
                    throw new AuraHandledException('Cannot change status to Rented: No active Rental Contract found for vehicle ' + vehicle.Registration_Number__c);
                }
            }
        }
    }
}
